# 01. Basic

기본 형태는 다음과 같다.

```jsx
import { AgGridReact } from "ag-grid-react";
import "ag-grid-enterprise/styles/ag-grid.css";
import "ag-grid-enterprise/styles/ag-theme-alpine.css";

export default function Default() {
  const columns = [{ field: "name" }, { field: "familyName" }];
  const rowDatas = [
    { name: "Hello", familyName: "World" },
    { name: "Lorem", familyName: "Ipsum" },
  ];

  return (
    <div style={{ height: "100%", boxSizing: "border-box" }}>
      <div
        style={{ width: "100%", height: 200 }}
        className="ag-theme-alpine"
      >
        <AgGridReact columnDefs={columns} rowData={rowDatas} />
      </div>
    </div>
  );
}
```

export const columns = () => [{ field: "name" }, { field: "familyName" }];
export const rowDatas = () => [
  { name: "Hello", familyName: "World" },
  { name: "Lorem", familyName: "Ipsum" },
];

<div style={{ height: "100%", boxSizing: "border-box" }}>

  >
    <AgGrid
      columnDefs={columns()}
      rowData={rowDatas()}
    />
  </div>
</div>
